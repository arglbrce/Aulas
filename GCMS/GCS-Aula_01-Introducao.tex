\begin{frame}
\frametitle{Mudanças}

\begin{itemize}
	\item \emph{Não há nada permanente, exceto a mudança}  (Heráclito, 500 a.C.)
	\item A maioria das modificações no software é justificável. 
	\item Em vez de evitá-las, devemos ter de mecanismos para cuidar delas.
\end{itemize}

\end{frame}

%------------------------------------------------

\begin{frame}
\frametitle{Mudanças}

\begin{itemize}
	\item Origens Comuns das Modificações:
	\begin{itemize}
		\item Novas condições de negócio ou mercado
		\item Novas necessidades do cliente
		\item Reorganização ou crescimento/diminuição dos negócios
		\item Restrições de orçamento ou cronograma
	\end{itemize}
\end{itemize}

\end{frame}

\begin{frame}
	\frametitle{O que é GCS?}

	\begin{itemize}
		\item GCS deve responder quem, o que, quando e por que:
		\begin{itemize}
			\item \textbf{Quem} fez as mudanças e podemos reproduzi-la?
			\begin{itemize}
				\item Auditoria responde
			\end{itemize}
		\end{itemize}
		\begin{itemize}
			\item \textbf{Que} mudanças e \textbf{quando} foram feitas no sistema?
			\begin{itemize}
				\item Controle de Versão responde
			\end{itemize}
		\end{itemize}		
		\begin{itemize}
			\item \textbf{Onde} as mudanças foram feitas?
			\begin{itemize}
				\item Controle de Versão responde
			\end{itemize}
		\end{itemize}
		\begin{itemize}
			\item \textbf{Por que} as mudanças foram feitas?
			\begin{itemize}
				\item Controle de Mudanças responde
			\end{itemize}
		\end{itemize}
	\end{itemize}

\end{frame}

\begin{frame}
	\frametitle{O que é GCS?}

	\begin{itemize}
		\item Sintomas da Falta de GCS:
		\begin{itemize}
			\item Não é possível determinar por que, quem e quando uma determinada alteração foi feita
			\item Não se sabe onde está a versão mais atual de um documento
			\item Falta de compatibilidade ou sincronismo entre diversos componentes de um projeto
		\end{itemize}
	\end{itemize}

\end{frame}

\begin{frame}
	\frametitle{O que é GCS?}

	\begin{itemize}
		\item Funções:
		\begin{itemize}
			\item \textbf{Identificação:} No início do desenvolvimento, a GCS permite à equipe de desenvolvimento identificar as unidades que compõem o sistema de acordo com as funcionalidades que elas deverão desempenhar.
			\item \textbf{Integração/Documentação:} As interfaces entre estas unidades, documentando assim a interação entre elas.
			\item \textbf{Controle:} O controle contínuo da evolução destas funcionalidades e interfaces permite que a integração entre estas unidades tenha sucesso continuado, com as mudanças devidamente gerenciadas e documentadas.
			\item \textbf{Auditoria:} A auditoria das funcionalidades identificadas, documentadas e controladas garante a confiabilidade do sistema.
		\end{itemize}
	\end{itemize}

\end{frame}

	\frametitle{Objetivo da GCS}

		\end{itemize}

\begin{frame}
	\frametitle{Abrangência de GCS}

	\begin{itemize}
		\item Gestão de configuração de software é um conjunto de atividades de planejamento, acompanhamento, controle e atividades técnicas que:
		\begin{itemize}
			\item Começa quando o projeto de engenharia de software tem início.
			\item Continua durante todas as versões posteriores do software.
			\item Só termina quando o software sai de produção.
		\end{itemize}
	\end{itemize}

\end{frame}
