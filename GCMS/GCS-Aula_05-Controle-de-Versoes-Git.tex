\begin{frame}
\frametitle{Características}

\begin{itemize}
	\item Controle de versão distribuído
	\item Padrão {\em de facto} atual
	\item Extremamente rápido nas operações
	\item Ótimo gerenciamento de merges e branches
\end{itemize}

\end{frame}

%------------------------------------------------

\begin{frame}
\frametitle{Características}

\begin{itemize}
	\item Snapshots, e não diferenças: fotografias
	\item Quase todas operações são locais: permite trabalho {\em offline}
	\item Tem Integridade: checksum SHA-1 string de 40 caracteres hexadecimais
	\item Geralmente só adiciona dados: quase impossível perder dados
	\item Três Estados: consolidado ({\em committed}), modificado ({\em modified}) e preparado ({\em staged})
\end{itemize}

\end{frame}

\begin{frame}
\frametitle{Armazenamento}

\begin{itemize}
	\item O Git armazena a nova versão de cada arquivo, e não o que mudou entre elas.
	\item Considera que os dados são como um conjunto de snapshots (uma foto) de um mini-sistema de arquivos.
\end{itemize}

\begin{frame}
\frametitle{Estados de arquivos}

\begin{itemize}
	\item Estados de arquivos:
	\begin{itemize}
		\item {\bf Consolidado} ({\em committed}): dados são ditos consolidados quando estão seguramente armazenados em sua base de dados local
		\item {\bf Modificado} ({\em modified}): Modificado trata de um arquivo que sofreu mudanças mas que ainda não foi consolidado na base de dados
		\item {\bf Preparado} ({\em staged}): Um arquivo está como preparado quando você marca o arquivo modificado em sua versão corrente para que ele faça parte do {\em snapshot} do próximo {\em commit} (consolidação)
	\end{itemize}
	\item Áreas de um projeto:
	\begin{itemize}
		\item Diretório do Git ({\em git directory, repository})
		\item Diretório de trabalho ({\em working directory})
		\item Área de preparação ({\em staging area}).
	\end{itemize}
\end{itemize}

\end{frame}

		\begin{itemize}
			\item {\bf leve}: {\tt git tag v1.4} ou {\tt git tag  v1.2 9fceb02}
			\begin{itemize}
				\item Similar a uma branch que não muda --- é um ponteiro para um commit específico.
			\end{itemize}
			
			\item {\bf anotada}: {\tt git tag -a v1.4 -m 'my version 1.4'}
			\begin{itemize}
				\item São objetos inteiros
				\item Chave de verificação
				\item Nome de quem criou a tag, email e data
				\item Mensagem relativa à tag
				\item Podem ser assinadas e verificadas com o GNU Privacy Guard (GPG)
			\end{itemize}
		\end{itemize}
	\end{itemize}

\begin{frame}[fragile]
	\frametitle{Rebasing}

\begin{itemize}
	\item Há duas forma de se integrar mudanças de branch para outro: {\tt merge} e {\tt rebase}
	\item Rebasing consiste em replicar todas as mudanças que foram realizadas num branch em outro branch
	\item Mantém o histórico mais limpo e linear (sem caminhos paralelos como no merge)
	\item O histórico aparecerá como se todo o trabalho tivesse sido em série quando na verdade foi realizado em paralelo
	\item Mantém a aparência do branch remoto/servidor mais limpa
\end{itemize}
\end{frame}

\item {\tt git remote -v}: lista o nome dos repositórios remotos e a URL
\item {\tt git remote add [shortname] [url]}: adiciona um repositório remoto ao repositório local
\item Exemplo: \\ {\tt git remote add pb https://github.com/paulboone/ticgit}
\item Exercício: adicionar o repositório do seu vizinho \\ {\tt git remote add vizinho https://github.com/vizinho/Spoon-Knife}
\item {\tt git remote rename pb paul}: renomeia um repositório remoto
\item {\tt git remote rm paul}: remove um repositório remoto
\item {\tt git remote show origin}: exibe detalhes do repositório remoto em relação ao repositório local. Quais branches estão sendo rastreadas

\end{itemize}

\end{frame}

\begin{frame}[fragile]
	\frametitle{{\tt git fetch}}

\begin{itemize}
\item {\tt git fetch [remote-name]}: 
	\begin{itemize}
		\item Recupera todos os dados remotos que não se tem ainda
		\item Referencia todos os branches remotos
	\end{itemize}
\item {\tt git fetch origin}: 
	\begin{itemize}
		\item Verifica qual servidor {\tt origin} aponta
		\item Obtém todos os dados que o repositório local ainda não tem, desde que foi feito {\tt git clone} ou {\tt git fetch}
		\item Atualiza o banco de dados local
		\item Move o  {\tt origin/master} para a posição mais recente e atualizada
		\item Não faz {\tt merge} automático
		\item As modificações locais ficam inalteradas até que se faça o {\tt merge} manual
	\end{itemize}
\end{itemize}

\end{frame}

\begin{itemize}
\item {\tt git pull}: Recupera  ({\tt fetch}) os dados do servidor que seu repositório foi clonado e tenta automaticamente realizar a fusão ({\tt merge}) com trabalho local
\end{itemize}

\begin{frame}[fragile]
	\frametitle{{\tt git push}}

	\begin{itemize}
	\item Enviando (Pushing): quando se quer compartilhar um branch enviando-o para um servidor remoto
	\item {\tt git push [remote-name] [branch-name]}
	\item {\tt git push origin master}
	\item Possui restrições:
	\begin{itemize}
		\item É preciso ter acesso de escrita no repositório remoto
		\item Ninguém realizou um {\tt git push} antes da sua tentativa
	\end{itemize}
	\item Exercício:
		\begin{itemize}
		\item Crie um branch local chamado {\tt teste-branch-remoto}
		\item Realize dois commits
		\item Entregue para o servidor: {\tt git push origin teste-branch-remoto}
		\item Visualize os resultados em modo gráfico
		\item No outro repositório local execute: {\tt git remote show origin}
	\end{itemize}
	\item {\tt git push origin --delete serverfix}: deleta um branch remoto
	\end{itemize}

\end{frame}
