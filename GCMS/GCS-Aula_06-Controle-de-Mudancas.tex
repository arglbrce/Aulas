\begin{frame}

\frametitle{Gestão de Mudanças}
\begin{itemize}
	\item Deve ser definido um documento/procedimento padrão para que mudanças possam ser solicitadas
	\item O artefato normalmente se chama Solicitação de Mudança (SM, em inglês CR)
	\item A um conjunto de pessoas (CCB), deve ser dada a autoridade para decidir se uma mudança será ou não implementada
	\item O processo é necessário para garantir que apenas mudanças avaliadas e aprovadas são realizadas
\end{itemize}

\end{frame}

\begin{frame}

\frametitle{Manutenção de Software}

\begin{itemize}
\item Mudanças efetuadas no software {\bf depois} de sua liberação
\item As alterações ocorrem por diversas razões, as quais determinam o {\bf tipo} de manutenção

\end{itemize}
\end{frame}

%------------------------------------------------

\begin{frame}

\frametitle{Tipos de Manutenção de Software}

	\begin{itemize}
	\item Manutenção Corretiva
		\begin{itemize}
			\item Identificar e corrigir erros
		\end{itemize}
	\item Manutenção Adaptativa
	\begin{itemize}
		\item Adaptar o software às mudanças ocorridas no ambiente (Normas, Leis, Mercado)
		\item Novas gerações de hardware
		\item Novos sistemas operacionais e software de apoio
		\item Atualizações e modificações em equipamentos periféricos e outros elementos de sistema
	\end{itemize}
	\end{itemize}
\end{frame}

%------------------------------------------------

\begin{frame}

\frametitle{Tipos de Manutenção de Software}

	\begin{itemize}
	\item Manutenção Melhoria/Evolutiva
		\begin{itemize}
			\item Modificar funções existentes
			\item Incluir novas funções (novos requisitos)
			\item Ampliar o escopo do software
			\item Efetuar melhorias em geral
		\end{itemize}
	\item Manutenção Preventiva
	\begin{itemize}
		\item Melhorar a manutenibilidade ou confiabilidade futuras e fornecer uma base melhor para futuros melhoramentos
	\end{itemize}
	\end{itemize}
\end{frame}

\begin{frame}

\frametitle{Gestão das Mudanças}

	\begin{itemize}
	\item As mudanças devem ser avaliadas antes de serem efetuadas
	\item Validade da requisição;
	\item Análise do impacto;
	\item Prioridade e estimativa de tempo;
	\item Responsável:  CCB do item
	\end{itemize}
\end{frame}

%------------------------------------------------

\begin{frame}

\frametitle{Gestão das Mudanças}

	\begin{itemize}
	\item O membro do CCB responsável por analisar a mudança deve:
	\begin{itemize}
		\item Analisar e documentar o impacto, consultando o cliente, se necessário;
		\item Rever a severidade e definir a prioridade.
	\end{itemize}
	\item Caso a mudança seja válida, designar um responsável para efetuá-la;
	\item Caso contrário, resolvê-la, informando a resolução tomada.
	\begin{itemize}
		\item Exemplos de resolução: inválida, duplicada, adiada.
	\end{itemize}
	\end{itemize}
\end{frame}

%------------------------------------------------

\begin{frame}

\frametitle{Gestão das Mudanças}

	\begin{itemize}
	\item A alteração dos itens de configuração deve seguir os seguintes passos:
	\begin{itemize}
		\item Realizar alteração;
		\item Atualizar item no repositório;
		\item Registrar histórico de alterações;
	\end{itemize}
	\item Se necessário pode ser gerada uma nova baseline.
	\end{itemize}
\end{frame}

%------------------------------------------------

\begin{frame}

\frametitle{Alterar Itens de Configuração}

	\begin{itemize}
	\item Realizar alteração
	\begin{itemize}
		\item Recuperar os itens a serem alterados, na versão correta, e realizar as alterações necessárias, em sua área de trabalho.
	\end{itemize}
	\end{itemize}
\end{frame}


%------------------------------------------------

\begin{frame}

\frametitle{Alterar Itens de Configuração}

	\begin{itemize}
	\item Atualizar item no repositório
	\begin{itemize}
		\item Após a alteração ter sido verificada, deve-se atualizar o item no repositório;
		\item No commit, deve-se informar o número da solicitação de mudança que motivou a alteração;
		\item Lembre de verificar se está trabalhando no branch correto!
	\end{itemize}
	\item Marcar a solicitação de mudança como resolvida, descrevendo que alterações foram efetuadas.
	\end{itemize}
\end{frame}
