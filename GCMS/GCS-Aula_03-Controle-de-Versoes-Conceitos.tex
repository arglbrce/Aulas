\begin{frame}
\frametitle{Ação: \emph{Checkout}}

	\begin{itemize}
		\item Algumas ferramentas permitem dois tipos de checkout:
		\begin{itemize}
			\item {\bf Reservado} (\emph{lock-modify-unlock})
			\begin{itemize}
				\item $\downarrow$ Somente um usuário terá a posse do arquivo, {\bf não permitindo} o trabalho simultâneo.
				\item $\uparrow$ Suas alterações irão constituir uma nova versão
			\end{itemize}
			\item {\bf Não Reservado} (\emph{copy-modify-merge})
			\begin{itemize}
				\item $\uparrow$ Vários usuários podem ter a posse do arquivo , {\bf permitindo} o trabalho simultâneo.
				\item $\downarrow$ Não é garantido que a sua versão será consolidada.
			\end{itemize}
		\end{itemize}
	\end{itemize}


\end{frame}

%------------------------------------------------

\begin{frame}
\frametitle{Ação: \emph{Checkin}}

	\begin{itemize}
		\item Gravar as alterações no repositório gerando uma nova versão do elemento
		\item Em algumas ferramentas chamado de \emph{Commit}
		\item Em casos de \emph{Checkout} Não-reservado pode gerar conflito de versões
	\end{itemize}

\begin{frame}
\frametitle{Ação: \emph{Merge} (Fusão)}

	\begin{itemize}
		\item Consiste em verificar as diferencças entre as alterações feitas por usuário distintos sobre o mesmo arquivo a aplicar a fusão de conteúdo gerando uma nova versão
	\end{itemize}

\begin{figure}
\includegraphics[scale=0.6]{./images/controle-versao-merge.png}
\end{figure}

\end{frame}

%------------------------------------------------

\begin{frame}
\frametitle{Ação: \emph{Merge} (Fusão)}

	\begin{itemize}
		\item Algumas ferramentas executam \emph{Merge} Automático em casos onde não há conflitos (alterações no mesmo trecho de código)
		\item Em caso de conflito o usuário terá que executar o \emph{merge} entre a versão dele e a versão remota (do repositório)
	\end{itemize}

\end{frame}

%------------------------------------------------
\begin{frame}
\frametitle{Ação: \emph{Branching} (Ramificação)}

\begin{itemize}
		\item Consiste em criar ramos paralelos de desenvolvimento (\emph{branches})
		\item Por default, existe pelo menos um branch em um projeto, o Principal, também conhecido como \emph{mainline} ou \emph{trunk} ou \emph{HEAD}.
	\end{itemize}

\end{frame}
